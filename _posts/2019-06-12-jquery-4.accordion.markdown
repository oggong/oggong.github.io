#### js/jquery.js

~~~

/**
 *
 */

$(document).ready(function() {

	$('.accordion').each(function() {

		// 변수 선언
		var allDt = $(this).find('dt');
		var allDd = $(this).find('dd');

		allDd.hide();
		allDt.css('cursor','pointer');

		allDt.click(function() {
			// alert('ok');
			allDd.slideUp('slow');
			$(this).next().slideDown('slow');
		});
	});

})

~~~
